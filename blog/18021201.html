ES6部分语法简述

# 列举常用的一些ES6语法

# 1.const 命令

  • const 声明一个只读的常量。一旦声明，常量的值就不能改变。不可重复声明
const PI = 3.14
PI  // 3.14
PI = 3  // TypeError: Assignment to constant variable
const PI = 3  // SyntaxError: Identifier 'PI' has already been declared
  • const命令声明的常量不提升，存在暂时性死区，只能在声明的位置后面使用
if (true) {
  console.log(PI)  // ReferenceError
  const PI = 3.14
}
  • 对于复合类型的变量 const 命令只是保证变量名指向的地址不变，并不保证该地址的数据不变
const foo = {}
foo.prop = 123
foo.prop  // 123
foo = {}  // TypeError: "foo" is read-only

# 2.let 命令

  • let命令所在的代码块内有效。不可重复声明
{
  let a = 10
  var b = 1
}
a  // ReferenceError: a is not defined.
b  // 1

for (let i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 0)
}  // 0, 1, 2, 3, 4
  • let命令声明的常量不提升，存在暂时性死区，只能在声明的位置后面使用
if (true) {
  console.log(PI)  // ReferenceError
  let PI = 3.14
}

# 3.变量的解构赋值

  • 从数组或对象中提取值，数组按照对应位置，对象按照相同的属性名,对变量赋值
var [a, b, c] = [1, 2]
a  // 1
b  // 2
c  // undefined

var { foo, bar } = { foo: 'aaa', bar: 'bbb' }
foo // "aaa"
bar // "bbb"

// 字符串被转换成了一个类似数组的对象
const [a, b, c, d, e] = 'hello'
a // "h"
b // "e"
c // "l"
d // "l"
e // "o"
  • 解构赋值允许指定默认值
let [a, b = 2] = [1]
a  // 1
b  // 2

# 4.模版字符串

  • 使用 ` 创建模版字符串。它可以当作普通字符串使用，也可以用来定义多行字符串，或者在字符串中嵌入变量(变量名写在${}之中)
const h = 'Hello'
const str = `${h} World!`
str  // "Hello World!"
  • 作为标签模板。紧跟在一个函数名后面，该函数将被调用来处理这个模板字符串
alert`123`
// 等同于
alert(123)

# 5.函数参数的默认值

  • ES6允许为函数的参数设置默认值
// es5
function hello (x, y) {
  y = y || 'World'
  console.log(x, y)
}

// es6
function hello (x, y = 'World') {
  console.log(x, y)
}
hello('Hello') // Hello World
hello('Hello', 'China') // Hello China
  • 参数默认值可以与解构赋值的默认值，结合起来使用
function hello ({x, y = 'World'}) {
  console.log(x, y)
}
hello({x: 'Hello'}) // Hello World
hello({x: 'Hello', y: 'China'}) // Hello China

function add ([a = 0, b = 0]) {
  return a + b
}
add([1, 2])  // 3
  • 函数参数的 rest 参数，将多余的参数放入数组中
function add(...values) {
  let sum = 0
  for (var val of values) {
    sum += val
  }
  return sum
}
add(2, 5, 3) // 10

# 6.扩展运算符

  • 扩展运算符 ... 它将一个数组转为用逗号分隔的参数序列
let arr = [2, 5, 3]

function add(...values) {
  let sum = 0
  for (var val of values) {
    sum += val
  }
  return sum
}
add(...arr)  // 10

Math.max(...arr)  // 5
  • 合并数组，生成数组
var arr1 = [0, 1, 2]
var arr2 = [3, 4, 5]
arr1.push(...arr2)
arr1  // [0, 1, 2, 3, 4, 5]

const [first, ...rest] = arr1
first // 1
rest // [2, 3, 4, 5]
  • 将字符串转为真正的数组
[...'hello']  // [ "h", "e", "l", "l", "o" ]

# 7.箭头函数

  • 使用 => 定义函数
var add = (a, b) => {
  a + b
}
// 相当于
var add = function (a, b) {
  return a + b
}
