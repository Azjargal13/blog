ES6 Promise对象

# Promise是异步编程的一种解决方案，能够很好的优化异步回调

  • Promise对象是一个构造函数，用来生成Promise实例
var promise = new Promise(function (resolve, reject) {
  // ... some code
  if (/* 异步操作成功 */){
    resolve(value)
  } else {
    reject(error)
  }
})
promise.then(function (value) {
  // ... some code
}).catch(function (error) {
  // ... some code
})
// 例子
var getJSON = function (url) {
  var promise = new Promise(function (resolve, reject){
    var client = new XMLHttpRequest()
    client.open("GET", url)
    client.onreadystatechange = handler
    client.responseType = "json"
    client.setRequestHeader("Accept", "application/json")
    client.send()
    function handler () {
      if (this.readyState !== 4) {
        return
      }
      if (this.status === 200) {
        resolve(this.response)
      } else {
        reject(new Error(this.statusText))
      }
    }
  })
  return promise
}

getJSON("/posts.json").then(function (json) {
  console.log('ok' + json)
}).catch(function (error) {
  console.log('error' + error)
})
  • Promise构造函数接受一个函数作为参数，该函数的两个参数分别是resolve和reject

# Promise.prototype.then()

  • then()调用后返回一个Promise对象，意味着实例化后的Promise对象可以进行链式调用，而且这个then()方法可以接收两个函数，一个是处理成功后的函数，一个是处理错误结果的函数(可选)
getJSON("/posts.json").then(function (json) {
  return json.post
}).then(function (post) {
  // ... some code
})

# Promise.prototype.catch()

  • 如果异步操作抛出错误，就会调用catch方法指定的回调函数。另外，then方法指定的回调函数运行中抛出错误，也会调用catch方法
getJSON("/posts.json").then(function (json) {
  return json.post
}).then(function (post) {
  // ... some code
}).catch(function (error) {
  // 处理 getJSON 和 前一个回调函数运行时发生的错误
  console.log('发生错误！', error)
})
  • catch方法返回的还是一个Promise对象，因此后面还可以接着调用then方法
getJSON("/posts.json").then(function (json) {
  return json.post
}).then(function (post) {
  // ... some code
}).catch(function (error) {
  // 处理 getJSON 和 前一个回调函数运行时发生的错误
  console.log('发生错误！', error)
}).then(function () {
  // ... some code
})

# Promise.all()

  • Promise.all方法用于将多个Promise实例，包装成一个新的Promise实例。接收一个参数，它必须是可以迭代的，比如数组
  • 用来处理一些并发的异步操作，即它们的结果互不干扰，但是又需要异步执行
  • 它的状态受参数内各个值的状态影响，即里面状态全部为fulfilled时，它才会变成fulfilled，否则变成rejected
var promises = [2, 3, 5, 7, 11, 13].map(function (id) {
  return getJSON(`/post/${id}.json`)
})
Promise.all(promises).then(function (posts) {
  // ... some code
}).catch(function (reason){
  // ... some code
})

# Promise.race()

  • Promise.race方法同样是将多个Promise实例，包装成一个新的Promise实例
  • 状态变化不是全部受参数内的状态影响，一旦参数内有一个值的状态发生的改变，那么该Promise的状态就是改变的状态
var test = function (id) {
  var promise = new Promise(function (resolve, reject){
    setTimeout(resolve, id * 1000, `${id} doned`)
  })
  return promise
}
var promises = [2, 3, 5, 7, 11, 13].map(function (id) {
  return test(id)
})
Promise.race(promises).then(function (value) {
  console.log(value)
}).catch(function (error){
  console.log(error)
})  // 2 doned
